JS:
let slideIndex = 1;
let slides;
let slideContainer;
let totalSlides;
let canSlide = true;
let slideWidth;

function initializeCarousel() {
  slides = document.getElementsByClassName("slide");
  slideContainer = document.querySelector(".slides");
  totalSlides = slides.length;

  if (totalSlides === 0) {
    console.error("No slides found. Please check your HTML structure.");
    return;
  }

  slideWidth = slideContainer.offsetWidth / 5; // Display 5 slides at a time

  // Set slide widths based on slide container width
  Array.from(slides).forEach((slide) => {
    slide.style.width = `${slideWidth}px`;
  });

  // Clone slides for infinite looping
  for (let i = 0; i < 2; i++) {
    const clonedSlides = Array.from(slides).map((slide) =>
      slide.cloneNode(true)
    );
    clonedSlides.forEach((clonedSlide) =>
      slideContainer.appendChild(clonedSlide)
    );
  }

  slideContainer.style.width = `${slideWidth * (totalSlides + 2)}px`; // Update the width to accommodate cloned slides
  slideContainer.style.transform = `translateX(-${slideWidth}px)`; // Start from the first slide

  // Handle transition end
  slideContainer.addEventListener("transitionend", handleTransitionEnd);
}

function changeSlide(n) {
  if (!canSlide) return;

  canSlide = false;
  slideIndex += n;

  slideContainer.style.transition = "transform 0.3s ease-in-out";
  slideContainer.style.transform = `translateX(-${slideWidth * slideIndex}px)`;
}

function handleTransitionEnd() {
  if (slideIndex === totalSlides + 1) {
    slideIndex = 1;
    slideContainer.style.transition = "none";
    slideContainer.style.transform = `translateX(-${slideWidth}px)`;
    setTimeout(() => {
      slideContainer.style.transition = "transform 0.3s ease-in-out";
      slideContainer.style.transform = `translateX(-${slideWidth * slideIndex}px)`;
    }, 0);
  } else if (slideIndex === 0) {
    slideIndex = totalSlides;
    slideContainer.style.transition = "none";
    slideContainer.style.transform = `translateX(-${slideWidth * totalSlides}px)`;
    setTimeout(() => {
      slideContainer.style.transition = "transform 0.3s ease-in-out";
      slideContainer.style.transform = `translateX(-${slideWidth * slideIndex}px)`;
    }, 0);
  }

  setTimeout(() => {
    canSlide = true;
  }, 100);
}

// Wait for the DOM to be loaded before initializing the carousel
document.addEventListener("DOMContentLoaded", initializeCarousel);

setInterval(() => {
  changeSlide(1);
}, 3000);
